var oldestYear = null; var oldestOption = null; // Loop through all the options in the dropdown $('#dateDropdown option').each(function() { // Get the value like '2001-2002' var value = $(this).val(); // Extract the beginning year (first part of the string) var yearBegin = parseInt(value.split('-')[0]); // Compare to find the oldest beginning year if (oldestYear === null || yearBegin < oldestYear) { oldestYear = yearBegin; oldestOption = $(this); } }); // Set the option with the oldest beginning year as selected if (oldestOption !== null) { oldestOption.prop('selected', true); }
